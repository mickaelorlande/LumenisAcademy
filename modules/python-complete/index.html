# Calculadora Neural - Lumenis Academy
# Projeto prático de Python com IA

import neural_calc
import utils

def main():
    print("🧠 Calculadora Neural Lumenis")
    print("=" * 40)
    
    while True:
        try:
            # Menu de opções
            print("\nEscolha uma operação:")
            print("1. Adição (+)")
            print("2. Subtração (-)")
            print("3. Multiplicação (*)")
            print("4. Divisão (/)")
            print("5. Análise Neural")
            print("0. Sair")
            
            opcao = input("\nSua escolha: ")
            
            if opcao == "0":
                print("Obrigado por usar a Calculadora Neural!")
                break
            elif opcao in ["1", "2", "3", "4"]:
                resultado = realizar_calculo(opcao)
                neural_calc.analisar_padrao(opcao, resultado)
            elif opcao == "5":
                neural_calc.mostrar_analise()
            else:
                print("Opção inválida! Tente novamente.")
                
        except KeyboardInterrupt:
            print("\n\nSaindo da calculadora...")
            break
        except Exception as e:
            print(f"Erro inesperado: {e}")

def realizar_calculo(opcao):
    operacoes = {
        "1": ("+", lambda x, y: x + y),
        "2": ("-", lambda x, y: x - y),
        "3": ("*", lambda x, y: x * y),
        "4": ("/", lambda x, y: x / y if y != 0 else "Erro: Divisão por zero!")
    }
    
    simbolo, funcao = operacoes[opcao]
    
    try:
        num1 = float(input("Digite o primeiro número: "))
        num2 = float(input("Digite o segundo número: "))
        
        resultado = funcao(num1, num2)
        
        print(f"\n🧮 Resultado: {num1} {simbolo} {num2} = {resultado}")
        
        # Explicação neural
        utils.explicar_processo_neural(simbolo, num1, num2, resultado)
        
        return resultado
        
    except ValueError:
        print("Erro: Digite apenas números!")
        return None
    except Exception as e:
        print(f"Erro no cálculo: {e}")
        return None

if __name__ == "__main__":
    main()

# Simula como o cérebro processa operações matemáticas

import time
from collections import defaultdict

# Armazena padrões de uso
padroes_uso = defaultdict(int)
historico_calculos = []

def analisar_padrao(operacao, resultado):
    """Analisa padrões neurais de uso da calculadora"""
    padroes_uso[operacao] += 1
    
    historico_calculos.append({
        'operacao': operacao,
        'resultado': resultado,
        'timestamp': time.time()
    })
    
    # Simula análise neural
    print(f"\n🧠 Análise Neural:")
    print(f"   Dopamina: +{calcular_dopamina(resultado)}%")
    print(f"   Acetilcolina: {calcular_foco()}%")
    print(f"   Padrão detectado: {detectar_padrao(operacao)}")

def calcular_dopamina(resultado):
    """Simula liberação de dopamina baseada no resultado"""
    if isinstance(resultado, (int, float)):
        if resultado > 0:
            return min(15, int(resultado / 10))
        else:
            return 5
    return 3

def calcular_foco():
    """Simula nível de foco baseado na atividade"""
    import random
    base_focus = 75
    variation = random.randint(-10, 15)
    return max(60, min(100, base_focus + variation))

def detectar_padrao(operacao):
    """Detecta padrões de preferência do usuário"""
    operacoes_nomes = {
        "1": "Adição",
        "2": "Subtração", 
        "3": "Multiplicação",
        "4": "Divisão"
    }
    
    mais_usada = max(padroes_uso.items(), key=lambda x: x[1], default=("1", 0))
    
    if mais_usada[1] > 3:
        return f"Preferência por {operacoes_nomes.get(mais_usada[0], 'Desconhecida')}"
    else:
        return "Explorando diferentes operações"

def mostrar_analise():
    """Mostra análise completa dos padrões neurais"""
    print("\n🔬 ANÁLISE NEURAL COMPLETA")
    print("=" * 50)
    
    if not historico_calculos:
        print("Nenhum dado disponível ainda. Faça alguns cálculos primeiro!")
        return
    
    print(f"📊 Total de operações: {len(historico_calculos)}")
    print(f"🎯 Operação favorita: {detectar_padrao('')}")
    
    # Análise temporal
    if len(historico_calculos) > 1:
        tempo_medio = calcular_tempo_medio()
        print(f"⏱️  Tempo médio entre operações: {tempo_medio:.1f}s")
        
        if tempo_medio < 10:
            print("   🚀 Velocidade alta - cérebro em modo foco!")
        elif tempo_medio < 30:
            print("   🧠 Velocidade normal - processamento equilibrado")
        else:
            print("   🤔 Velocidade reflexiva - análise profunda")
    
    print(f"\n💡 Recomendação Neural:")
    gerar_recomendacao()

def calcular_tempo_medio():
    """Calcula tempo médio entre operações"""
    if len(historico_calculos) < 2:
        return 0
    
    tempos = []
    for i in range(1, len(historico_calculos)):
        tempo_diff = historico_calculos[i]['timestamp'] - historico_calculos[i-1]['timestamp']
        tempos.append(tempo_diff)
    
    return sum(tempos) / len(tempos)

def gerar_recomendacao():
    """Gera recomendações baseadas em padrões neurais"""
    total_ops = len(historico_calculos)
    
    if total_ops < 5:
        print("   Continue praticando para desenvolver padrões neurais!")
    elif total_ops < 15:
        print("   Seu cérebro está criando conexões! Tente operações mais complexas.")
    else:
        print("   Excelente! Você está desenvolvendo fluência matemática neural.")
        print("   Recomendo avançar para funções e estruturas de dados!")

def gerar_exercicio_personalizado(nivel_atual):
    """Gera exercícios baseados no nível neural atual"""
    import random
    
    if nivel_atual < 50:
        # Exercícios básicos
        num1 = random.randint(1, 20)
        num2 = random.randint(1, 20)
        operacao = random.choice(['+', '-'])
    elif nivel_atual < 80:
        # Exercícios intermediários
        num1 = random.randint(10, 100)
        num2 = random.randint(10, 100)
        operacao = random.choice(['+', '-', '*'])
    else:
        # Exercícios avançados
        num1 = random.randint(50, 500)
        num2 = random.randint(10, 50)

def calcular_neuroplasticidade(historico_performance):
    """Calcula índice de neuroplasticidade baseado na performance"""
    if len(historico_performance) < 3:
        return 50  # Valor inicial
    
    # Analisa tendência de melhoria
    recentes = historico_performance[-5:]  # Últimas 5 performances
    antigas = historico_performance[-10:-5] if len(historico_performance) >= 10 else historico_performance[:-5]
    
    if not antigas:
        return 60
    
    media_recente = sum(recentes) / len(recentes)
    media_antiga = sum(antigas) / len(antigas)
    
    melhoria = ((media_recente - media_antiga) / media_antiga) * 100
    
    # Índice base + melhoria
    indice = 50 + min(50, max(-20, melhoria))
    
    return round(indice, 1)

# Funções de gamificação
def calcular_pontos_xp(operacao, tempo_resposta, acertos_consecutivos):
    """Calcula pontos XP baseados na performance"""
    pontos_base = {
        "+": 10,
        "-": 10, 
        "*": 15,
        "/": 20
    }
    
    pontos = pontos_base.get(operacao, 10)
    
    # Bônus por velocidade
    if tempo_resposta < 5:
        pontos += 5
    elif tempo_resposta < 10:
        pontos += 2
    
    # Bônus por sequência
    if acertos_consecutivos > 3:
        pontos += acertos_consecutivos * 2
    
    return pontos
